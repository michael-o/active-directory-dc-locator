16  package net.sf.michaelo.activedirectory.ldapping;
17  
18  import net.sf.michaelo.activedirectory.IntFlag;
19  
20  /**
21   * A class representing the <a href=
22   * "https://learn.microsoft.com/en-us/openspecs/windows_protocols/ms-adts/f55d3f53-351d-4407-940e-f53eb6154af0"><code>
23   * DS_FLAG</code></a> options bits.
24   */
25  public enum DsFlag implements IntFlag {
26  	FP(0x00000001, "DS_PDC_FLAG"),
27  	FG(0x00000004, "DS_GC_FLAG"),
28  	FL(0x00000008, "DS_LDAP_FLAG"),
29  	FD(0x00000010, "DS_DS_FLAG"),
30  	FK(0x00000020, "DS_KDC_FLAG"),
31  	FT(0x00000040, "DS_TIMESERV_FLAG"),
32  	FC(0x00000080, "DS_CLOSEST_FLAG"),
33  	FW(0x00000100, "DS_WRITABLE_FLAG"),
34  	FGT(0x00000200, "DS_GOOD_TIMESERV_FLAG"),
35  	FN(0x00000400, "DS_NDNC_FLAG"),
36  	FSS(0x00000800, "DS_SELECT_SECRET_DOMAIN_6_FLAG"),
37  	FFS(0x00001000, "DS_FULL_SECRET_DOMAIN_6_FLAG"),
38  	FWS(0x00002000, "DS_WS_FLAG"),
39  	FW8(0x00004000, "DS_DS_8_FLAG"),
40  	FW9(0x00008000, "DS_DS_9_FLAG"),
41  	FW10(0x00010000, "DS_DS_10_FLAG"),
42  	// This is not documented: https://nettools.net/nltest-flags-what-does-0x20000-mean/
43  	FKL(0x00020000, "DS_KEY_LIST_FLAG"),
44  	// Flags returned on ping: int DS_PING_FLAGS = 0x000FFFFF
45  	FDNS(0x20000000, "DS_DNS_CONTROLLER_FLAG"),
46  	FDM(0x40000000, "DS_DNS_DOMAIN_FLAG"),
47  	FF(0x80000000, "DS_DNS_FOREST_FLAG");
48  
49  	private final int intValue;
50  	private final String fullName;
51  
52  	DsFlag(int intValue, String fullName) {
53  		this.intValue = intValue;
54  		this.fullName = fullName;
55  	}
56  
57  	public int intValue() {
58  		return intValue;
59  	}
60  
61  	public String fullName() {
62  		return fullName;
63  	}
64  }
