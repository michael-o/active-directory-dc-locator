16  package net.sf.michaelo.activedirectory.ldapping;
17  
18  import net.sf.michaelo.activedirectory.IntFlag;
19  
20  /**
21   * A class representing the <a href=
22   * "https://learn.microsoft.com/en-us/openspecs/windows_protocols/ms-adts/8e6a9efa-6312-44e2-af12-06ad73afbfa5"><code>
23   * NETLOGON_NT_VERSION</code></a> options bits.
24   */
25  public enum NetlogonNtVersion implements IntFlag {
26  	V1(0x00000001, "NETLOGON_NT_VERSION_1"),
27  	V5(0x00000002, "NETLOGON_NT_VERSION_5"),
28  	V5EX(0x00000004, "NETLOGON_NT_VERSION_5EX"),
29  	V5EP(0x00000008, "NETLOGON_NT_VERSION_5EX_WITH_IP"),
30  	VCS(0x00000010, "NETLOGON_NT_VERSION_WITH_CLOSEST_SITE"),
31  	VNT4(0x01000000, "NETLOGON_NT_VERSION_AVOID_NT4EMUL"),
32  	VPDC(0x10000000, "NETLOGON_NT_VERSION_PDC"),
33  	VIP(0x20000000, "NETLOGON_NT_VERSION_IP"),
34  	VL(0x40000000, "NETLOGON_NT_VERSION_LOCAL"),
35  	VGC(0x80000000, "NETLOGON_NT_VERSION_GC");
36  
37  	private final int intValue;
38  	private final String fullName;
39  
40  	NetlogonNtVersion(int intValue, String fullName) {
41  		this.intValue = intValue;
42  		this.fullName = fullName;
43  	}
44  
45  	public int intValue() {
46  		return intValue;
47  	}
48  
49  	public String fullName() {
50  		return fullName;
51  	}
52  }
