34  /** Base class for all Netlogon SAM Logon structures. */
35  public abstract class NetlogonSamLogonBaseResponse {
36  
37  	private static final Logger LOGGER = Logger.getLogger(NetlogonSamLogonBaseResponse.class.getName());
38  
39  	private static final byte LABEL_NORMAL = (byte) 0;
40  	private static final byte LABEL_COMPRESSION = (byte) 0xC0;
41  	private static final byte LABEL_MASK = (byte) 0xC0;
42  
43  	protected Opcode opcode;
44  	protected Set<NetlogonNtVersion> ntVersion;
45  
46  	protected ByteBuffer wrap(byte[] netlogonBytes) {
47  		Objects.requireNonNull(netlogonBytes, "netlogonBytes cannot be null");
48  		if (netlogonBytes.length == 0) throw new IllegalArgumentException("netlogonBytes cannot be empty");
49  
50  		ByteBuffer buf = ByteBuffer.wrap(netlogonBytes);
51  		buf.order(ByteOrder.LITTLE_ENDIAN);
52  
53  		return buf;
54  	}
55  
56  	protected Opcode getOpcode(ByteBuffer buf, Opcode expectedOpcode) {
57  		Opcode opcode = Opcode.fromShortValue(buf.getShort());
58  		if (opcode != expectedOpcode)
59  			throw new IllegalArgumentException("Opcode must be " + expectedOpcode + ", but is " + opcode);
60  		return opcode;
61  	}
62  
63  	protected Set<NetlogonNtVersion> getNtVersion(ByteBuffer buf, Set<NetlogonNtVersion> expectedNtVersion) {
64  		Set<NetlogonNtVersion> ntVersion = IntFlag.fromFlags(NetlogonNtVersion.class, buf.getInt());
65  		if (!ntVersion.containsAll(expectedNtVersion))
66  			throw new IllegalArgumentException("NtVersion must contain [" + IntFlag.toFlagsString(expectedNtVersion)
67  					+ "], but contains [" + IntFlag.toFlagsString(ntVersion) + "]");
68  		return ntVersion;
69  	}
70  
71  	protected String getUnicodeString(ByteBuffer buf) {
72  		if (buf.remaining() < 2)
73  			throw new IllegalArgumentException("Buffer has not enough bytes to read Unicode string");
74  		int startPos = ((Buffer) buf).position();
75  		int endPos = -1;
76  		boolean nullTerminatorFound = false;
77  		while (buf.hasRemaining() && !nullTerminatorFound) {
78  			byte low = buf.get();
79  			byte high = buf.get();
80  			endPos = ((Buffer) buf).position();
81  			if (low == 0x00 && high == 0x00) nullTerminatorFound = true;
82  		}
83  		if (!nullTerminatorFound) throw new IllegalArgumentException("Buffer contains unterminated Unicode string");
84  		byte[] strBytes = new byte[endPos - startPos];
85  		((Buffer) buf).position(startPos);
86  		buf.get(strBytes);
87  		if (strBytes.length == 2 && strBytes[0] == 0x00 && strBytes[1] == 0x00) return null;
88  		return new String(strBytes, 0, strBytes.length - 2 /* delete null */, StandardCharsets.UTF_16LE);
89  	}
90  
91  	protected UUID getGuid(ByteBuffer buf) {
92  		// https://stackoverflow.com/a/28628209/696632
93  		ByteBuffer guidBuf = ByteBuffer.allocate(16);
94  		ByteOrder order = buf.order();
95  		buf.order(ByteOrder.LITTLE_ENDIAN);
96  		guidBuf.putInt(buf.getInt()).putShort(buf.getShort()).putShort(buf.getShort());
97  		buf.order(ByteOrder.BIG_ENDIAN);
98  		guidBuf.putLong(buf.getLong());
99  		((Buffer) guidBuf).rewind();
100 		buf.order(order);
101 		return new UUID(guidBuf.getLong(), guidBuf.getLong());
102 	}
103 
104 	protected String getCompressedDomainName(ByteBuffer buf) {
105 		boolean done = false;
106 		boolean pointerMode = false;
107 		List<String> labels = new ArrayList<>();
108 		int currPos = ((Buffer) buf).position();
109 		while (buf.hasRemaining() && !done) {
110 			byte len = buf.get();
111 			if (!pointerMode) currPos++;
112 			switch (len & LABEL_MASK) {
113 				case LABEL_NORMAL:
114 					if (len == 0) {
115 						done = true;
116 					} else {
117 						if (len > buf.remaining())
118 							throw new IllegalArgumentException(String.format(
119 									"Domain label longer (%d) than bytes in buffer (%d)", len, buf.remaining()));
120 						byte[] labelBytes = new byte[len];
121 						buf.get(labelBytes);
122 						labels.add(new String(labelBytes, StandardCharsets.UTF_8));
123 						if (!pointerMode) currPos += len;
124 					}
125 					break;
126 				case LABEL_COMPRESSION:
127 					short offset = (short) ((len & ~LABEL_COMPRESSION) << 8 | buf.get() & 0xFF);
128 					if (!pointerMode) currPos++;
129 					// position of pointer minus two length bytes
130 					int maxOffset = currPos - 2;
131 					if (offset >= maxOffset)
132 						throw new IllegalArgumentException(String.format(
133 								"Domain name pointer offset (%d) beyond current position (%d)", offset, maxOffset));
134 					((Buffer) buf).position(offset);
135 					pointerMode = true;
136 					break;
137 				default:
138 					throw new IllegalArgumentException(String.format("Invalid domain name label type: 0x%02X", len));
139 			}
140 		}
141 		((Buffer) buf).position(currPos);
142 
143 		if (labels.isEmpty()) return null;
144 
145 		return String.join(".", labels);
146 	}
147 
148 	protected InetAddress getInetAddress(byte[] addressBytes) {
149 		try {
150 			return InetAddress.getByAddress(addressBytes);
151 		} catch (UnknownHostException e) {
152 			LOGGER.fine(() ->
153 					String.format("Failed to convert address bytes %s to InetAddress", Arrays.toString(addressBytes)));
154 			return null;
155 		}
156 	}
157 
158 	protected void consumeLmTokens(ByteBuffer buf) {
159 		short lmNtToken = buf.getShort();
160 		if ((lmNtToken & 0xFFFF) == 0)
161 			throw new IllegalArgumentException(String.format("LmNtToken must be 0xFFFF, but is 0x%04X", lmNtToken));
162 		short lm20Token = buf.getShort();
163 		if ((lm20Token & 0xFFFF) == 0)
164 			throw new IllegalArgumentException(String.format("Lm20Token must be 0xFFFF, but is 0x%04X", lm20Token));
165 	}
166 
167 	public Opcode getOpcode() {
168 		return opcode;
169 	}
170 
171 	public Set<NetlogonNtVersion> getNtVersion() {
172 		return ntVersion;
173 	}
174 }
