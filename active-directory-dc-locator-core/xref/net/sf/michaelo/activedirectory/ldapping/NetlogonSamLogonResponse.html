16  package net.sf.michaelo.activedirectory.ldapping;
17  
18  import java.net.InetAddress;
19  import java.nio.ByteBuffer;
20  import java.util.Collections;
21  import java.util.EnumSet;
22  import java.util.Set;
23  import java.util.UUID;
24  
25  import net.sf.michaelo.activedirectory.IntFlag;
26  import net.sf.michaelo.activedirectory.StringUtils;
27  
28  /**
29   * A class representing the <a href=
30   * "https://learn.microsoft.com/en-us/openspecs/windows_protocols/ms-adts/9b333971-6dd7-4003-a898-6e51b2c02110"><code>
31   * NETLOGON_SAM_LOGON_RESPONSE</code></a> structure.
32   */
33  public class NetlogonSamLogonResponse extends NetlogonSamLogonBaseResponse {
34  
35  	private static final Set<NetlogonNtVersion> NETLOGON_NT_VERSION_5 =
36  			Collections.unmodifiableSet(EnumSet.of(NetlogonNtVersion.V1, NetlogonNtVersion.V5));
37  
38  	private UUID domainGuid;
39  	private String unicodeLogonServer;
40  	private String unicodeUserName;
41  	private String unicodeDomainName;
42  	private String dnsForestName;
43  	private String dnsDomainName;
44  	private String dnsHostName;
45  	private InetAddress dcIpAddress;
46  	private Set<DsFlag> flags;
47  
48  	public NetlogonSamLogonResponse(byte[] netlogonBytes) {
49  		ByteBuffer buf = wrap(netlogonBytes);
50  
51  		// Read NETLOGON_SAM_LOGON_RESPONSE structure
52  		// Opcode
53  		this.opcode = getOpcode(buf, Opcode.LOGON_SAM_LOGON_RESPONSE);
54  		// UnicodeLogonServer
55  		this.unicodeLogonServer = getUnicodeString(buf);
56  		// UnicodeUserName
57  		this.unicodeUserName = getUnicodeString(buf);
58  		// UnicodeDomainName
59  		this.unicodeDomainName = getUnicodeString(buf);
60  		// DomainGuid
61  		this.domainGuid = getGuid(buf);
62  		// NullGuid
63  		buf.getLong();
64  		buf.getLong();
65  		// DnsForestName
66  		dnsForestName = getCompressedDomainName(buf);
67  		// DnsDomainName
68  		dnsDomainName = getCompressedDomainName(buf);
69  		// DnsHostName
70  		dnsHostName = getCompressedDomainName(buf);
71  		// DcIpAddress
72  		byte[] dcIpAddressBytes = new byte[4];
73  		// swap byte order since Windows uses little endian, but everyone else big endian
74  		dcIpAddressBytes[3] = buf.get();
75  		dcIpAddressBytes[2] = buf.get();
76  		dcIpAddressBytes[1] = buf.get();
77  		dcIpAddressBytes[0] = buf.get();
78  		dcIpAddress = getInetAddress(dcIpAddressBytes);
79  		// Flags
80  		this.flags = IntFlag.fromFlags(DsFlag.class, buf.getInt());
81  		// NtVersion
82  		this.ntVersion = getNtVersion(buf, NETLOGON_NT_VERSION_5);
83  		// LmNtToken + Lm20Token
84  		consumeLmTokens(buf);
85  	}
86  
87  	public String getUnicodeLogonServer() {
88  		return unicodeLogonServer;
89  	}
90  
91  	public String getUnicodeUserName() {
92  		return unicodeUserName;
93  	}
94  
95  	public String getUnicodeDomainName() {
96  		return unicodeDomainName;
97  	}
98  
99  	public UUID getDomainGuid() {
100 		return domainGuid;
101 	}
102 
103 	public String getDnsForestName() {
104 		return dnsForestName;
105 	}
106 
107 	public String getDnsDomainName() {
108 		return dnsDomainName;
109 	}
110 
111 	public String getDnsHostName() {
112 		return dnsHostName;
113 	}
114 
115 	public InetAddress getDcIpAddress() {
116 		return dcIpAddress;
117 	}
118 
119 	public Set<DsFlag> getFlags() {
120 		return flags;
121 	}
122 
123 	@Override
124 	public String toString() {
125 		StringBuilder sb = new StringBuilder("NetlogonSamLogonResponse[");
126 		sb.append("opcode: ").append(opcode);
127 		if (StringUtils.isNotEmpty(unicodeLogonServer))
128 			sb.append(", unicodeLogonServer: ").append(unicodeLogonServer);
129 		if (StringUtils.isNotEmpty(unicodeUserName))
130 			sb.append(", unicodeUserName: ").append(unicodeUserName);
131 		if (StringUtils.isNotEmpty(unicodeDomainName))
132 			sb.append(", unicodeDomainName: ").append(unicodeDomainName);
133 		sb.append(", domainGuid: ").append(domainGuid);
134 		sb.append(", dnsForestName: ").append(dnsForestName);
135 		sb.append(", dnsDomainName: ").append(dnsDomainName);
136 		sb.append(", dnsHostName: ").append(dnsHostName);
137 		sb.append(", dcIpAddress: ").append(dcIpAddress.getHostAddress());
138 		sb.append(", flags: ").append(IntFlag.toFlagsString(flags));
139 		sb.append(", ntVersion: ").append(IntFlag.toFlagsString(ntVersion));
140 		sb.append("]");
141 		return sb.toString();
142 	}
143 }
