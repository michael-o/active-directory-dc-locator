36  /**
37   * An LDAP ping {@code NetlogonSamLogonBaseResponse} subclasses dump printer produced by
38   * {@link ActiveDirectoryLdapPingerTester}.
39   *
40   * <p>This class can be called via its main method, it supports the following optional parameters:
41   *
42   * <ul>
43   *   <li>output format {@code --format} {@code listing} (default) or {@code sql},
44   * </ul>
45   *
46   * <p>and the following positional parameters:
47   *
48   * <ul>
49   *   <li>dump file {@code path...}: a file containing dumps.
50   * </ul>
51   *
52   * <p>The {@code sql} format output can be used to import the data into a SQLite database for later analysis.
53   */
54  public class ActiveDirectoryLdapPingerDumpPrinter {
55  
56  	private static final AtomicInteger NETLOGON_ID_GENERATOR = new AtomicInteger();
57  
58  	private static void dumpFile(Path file, String format) throws IOException {
59  		System.err.printf("Processing file '%s'%n", file);
60  
61  		try (Stream<String> lines = Files.lines(file)) {
62  			Iterator<String> iter = lines.iterator();
63  			while (iter.hasNext()) {
64  				String[] cols = iter.next().split(";", 6);
65  				String domain = cols[0];
66  				String host = cols[1];
67  				String dnsDomain = cols[2];
68  				String dnsHostName = cols[3];
69  				String ntVersionStr = cols[4];
70  				Set<NetlogonNtVersion> ntVersion = IntFlag.fromFlagsString(NetlogonNtVersion.class, ntVersionStr);
71  				String output = cols[5];
72  
73  				int requestId = NETLOGON_ID_GENERATOR.incrementAndGet();
74  
75  				switch (format) {
76  					case "listing":
77  						System.out.println("Request:");
78  						System.out.println("  domain: " + domain);
79  						System.out.println("  host: " + host);
80  						if (!dnsDomain.isEmpty()) System.out.println("  dnsDomain: " + dnsDomain);
81  						if (!dnsHostName.isEmpty()) System.out.println("  dnsHostName: " + dnsHostName);
82  						System.out.println("  ntVersion: " + ntVersionStr);
83  						break;
84  					case "sql":
85  						System.out.printf(
86  								"insert into netlogon_request(id, domain, host, dnsDomain, dnsHostName, ntVersion)"
87  										+ " values(%d, '%s', '%s', %s, %s, '%s');%n",
88  								requestId, domain, host, nullSafe(dnsDomain), nullSafe(dnsHostName), ntVersionStr);
89  						break;
90  				}
91  
92  				switch (format) {
93  					case "listing":
94  						System.out.println("Response:");
95  						break;
96  					case "sql":
97  						System.out.printf(
98  								"insert into netlogon_response_info(requestId, responseType, exception)"
99  										+ " values(%d, ",
100 								requestId);
101 						break;
102 				}
103 				if (output.isEmpty()) {
104 					switch (format) {
105 						case "listing":
106 							System.out.println("  (none)");
107 							break;
108 						case "sql":
109 							System.out.println("NULL, NULL);");
110 							break;
111 					}
112 				} else {
113 					if (output.startsWith("exception:")) {
114 						String exception = output.substring("exception:".length());
115 						switch (format) {
116 							case "listing":
117 								System.out.println("  " + exception);
118 								break;
119 							case "sql":
120 								System.out.printf("NULL, '%s');%n", exception);
121 								break;
122 						}
123 					} else {
124 						byte[] netlogon = Base64.getDecoder().decode(output.substring("base64:".length()));
125 
126 						if (ntVersion.contains(NetlogonNtVersion.V5EX)) {
127 							NetlogonSamLogonExResponse response = new NetlogonSamLogonExResponse(netlogon);
128 							switch (format) {
129 								case "listing":
130 									System.out.println("  NetlogonSamLogonExResponse:");
131 									break;
132 								case "sql":
133 									System.out.println("'NETLOGON_SAM_LOGON_RESPONSE_EX', NULL);");
134 									break;
135 							}
136 							Opcode opcode = response.getOpcode();
137 							String opcodeStr = String.valueOf(opcode);
138 							Set<DsFlag> flags = response.getFlags();
139 							String flagsStr = IntFlag.toFlagsString(flags);
140 							UUID domainGuid = response.getDomainGuid();
141 							String dnsForestName = response.getDnsForestName();
142 							String dnsDomainName = response.getDnsDomainName();
143 							dnsHostName = response.getDnsHostName();
144 							String netbiosDomainName = response.getNetbiosDomainName();
145 							String netbiosComputerName = response.getNetbiosComputerName();
146 							String userName = response.getUserName();
147 							String dcSiteName = response.getDcSiteName();
148 							String clientSiteName = response.getClientSiteName();
149 							String dcSockAddr = response.getDcSockAddr() != null
150 									? response.getDcSockAddr().getHostAddress()
151 									: null;
152 							String nextClosestSiteName = response.getNextClosestSiteName();
153 							ntVersion = response.getNtVersion();
154 							ntVersionStr = IntFlag.toFlagsString(ntVersion);
155 							switch (format) {
156 								case "listing":
157 									System.out.println("    opcode: " + opcodeStr);
158 									System.out.println("    flags: " + flagsStr);
159 									System.out.println("    domainGuid: " + domainGuid);
160 									System.out.println("    dnsForestName: " + dnsForestName);
161 									System.out.println("    dnsDomainName: " + dnsDomainName);
162 									System.out.println("    dnsHostName: " + dnsHostName);
163 									if (netbiosDomainName != null)
164 										System.out.println("    netbiosDomainName: " + netbiosDomainName);
165 									if (netbiosComputerName != null)
166 										System.out.println("    netbiosComputerName: " + netbiosComputerName);
167 									if (userName != null) System.out.println("    userName: " + userName);
168 									System.out.println("    dcSiteName: " + dcSiteName);
169 									if (clientSiteName != null)
170 										System.out.println("    clientSiteName: " + clientSiteName);
171 									if (dcSockAddr != null) System.out.println("    dcSockAddr: " + dcSockAddr);
172 									if (nextClosestSiteName != null)
173 										System.out.println("    nextClosestSiteName: " + nextClosestSiteName);
174 									System.out.println("    ntVersion: " + ntVersionStr);
175 									break;
176 								case "sql":
177 									System.out.printf(
178 											"insert into netlogon_sam_logon_ex_response(requestId, opcode, flags, domainGuid,"
179 													+ " dnsForestName, dnsDomainName, dnsHostName, netbiosDomainName, netbiosComputerName,"
180 													+ " userName, dcSiteName, clientSiteName, dcSockAddr, nextClosestSiteName, ntVersion)"
181 													+ " values(%d, '%s', '%s', '%s', '%s', '%s', '%s', %s, %s, %s, '%s', %s, %s, %s, '%s');%n",
182 											requestId,
183 											opcodeStr,
184 											flagsStr,
185 											domainGuid,
186 											dnsForestName,
187 											dnsDomainName,
188 											dnsHostName,
189 											nullSafe(netbiosDomainName),
190 											nullSafe(netbiosComputerName),
191 											nullSafe(userName),
192 											dcSiteName,
193 											nullSafe(clientSiteName),
194 											nullSafe(dcSockAddr),
195 											nullSafe(nextClosestSiteName),
196 											ntVersionStr);
197 									break;
198 							}
199 
200 						} else if (ntVersion.contains(NetlogonNtVersion.V5)) {
201 							NetlogonSamLogonResponse response = new NetlogonSamLogonResponse(netlogon);
202 							switch (format) {
203 								case "listing":
204 									System.out.println("  NetlogonSamLogonResponse:");
205 									break;
206 								case "sql":
207 									System.out.println("'NETLOGON_SAM_LOGON_RESPONSE', NULL);");
208 									break;
209 							}
210 							Opcode opcode = response.getOpcode();
211 							String opcodeStr = String.valueOf(opcode);
212 							String unicodeLogonServer = response.getUnicodeLogonServer();
213 							String unicodeUserName = response.getUnicodeUserName();
214 							String unicodeDomainName = response.getUnicodeDomainName();
215 							UUID domainGuid = response.getDomainGuid();
216 							String dnsForestName = response.getDnsForestName();
217 							String dnsDomainName = response.getDnsDomainName();
218 							dnsHostName = response.getDnsHostName();
219 							String dcIpAddress = response.getDcIpAddress().getHostAddress();
220 							Set<DsFlag> flags = response.getFlags();
221 							String flagsStr = IntFlag.toFlagsString(flags);
222 							ntVersion = response.getNtVersion();
223 							ntVersionStr = IntFlag.toFlagsString(ntVersion);
224 							switch (format) {
225 								case "listing":
226 									System.out.println("    opcode: " + opcodeStr);
227 									if (unicodeLogonServer != null)
228 										System.out.println("    unicodeLogonServer: " + unicodeLogonServer);
229 									if (unicodeUserName != null)
230 										System.out.println("    unicodeUserName: " + unicodeUserName);
231 									if (unicodeDomainName != null)
232 										System.out.println("    unicodeDomainName: " + unicodeDomainName);
233 									System.out.println("    domainGuid: " + domainGuid);
234 									System.out.println("    dnsForestName: " + dnsForestName);
235 									System.out.println("    dnsDomainName: " + dnsDomainName);
236 									System.out.println("    dnsHostName: " + dnsHostName);
237 									System.out.println("    dcIpAddress: " + dcIpAddress);
238 									System.out.println("    flags: " + flagsStr);
239 									System.out.println("    ntVersion: " + ntVersionStr);
240 									break;
241 								case "sql":
242 									System.out.printf(
243 											"insert into netlogon_sam_logon_response(requestId, opcode, unicodeLogonServer, unicodeUserName,"
244 													+ " unicodeDomainName, domainGuid, dnsForestName, dnsDomainName, dnsHostName,"
245 													+ " dcIpAddress, flags, ntVersion)"
246 													+ " values(%d, '%s', %s, %s, %s, '%s', '%s', '%s', '%s', '%s', '%s', '%s');%n",
247 											requestId,
248 											opcodeStr,
249 											nullSafe(unicodeLogonServer),
250 											nullSafe(unicodeUserName),
251 											nullSafe(unicodeDomainName),
252 											domainGuid,
253 											dnsForestName,
254 											dnsDomainName,
255 											dnsHostName,
256 											dcIpAddress,
257 											flagsStr,
258 											ntVersionStr);
259 									break;
260 							}
261 						} else {
262 							NetlogonSamLogonNt40Response response = new NetlogonSamLogonNt40Response(netlogon);
263 							switch (format) {
264 								case "listing":
265 									System.out.println("  NetlogonSamLogonNt40Response:");
266 									break;
267 								case "sql":
268 									System.out.println("'NETLOGON_SAM_LOGON_RESPONSE_NT40', NULL);");
269 									break;
270 							}
271 							Opcode opcode = response.getOpcode();
272 							String opcodeStr = String.valueOf(opcode);
273 							String unicodeLogonServer = response.getUnicodeLogonServer();
274 							String unicodeUserName = response.getUnicodeUserName();
275 							String unicodeDomainName = response.getUnicodeDomainName();
276 							ntVersion = response.getNtVersion();
277 							ntVersionStr = IntFlag.toFlagsString(ntVersion);
278 							switch (format) {
279 								case "listing":
280 									System.out.println("    opcode: " + opcodeStr);
281 									if (unicodeLogonServer != null)
282 										System.out.println("    unicodeLogonServer: " + unicodeLogonServer);
283 									if (unicodeUserName != null)
284 										System.out.println("    unicodeUserName: " + unicodeUserName);
285 									if (unicodeDomainName != null)
286 										System.out.println("    unicodeDomainName: " + unicodeDomainName);
287 									System.out.println("    ntVersion: " + ntVersionStr);
288 									break;
289 								case "sql":
290 									System.out.printf(
291 											"insert into netlogon_sam_logon_nt40_response(requestId, opcode, unicodeLogonServer, unicodeUserName,"
292 													+ " unicodeDomainName, ntVersion)"
293 													+ " values(%d, '%s', %s, %s, %s, '%s');%n",
294 											requestId,
295 											opcodeStr,
296 											nullSafe(unicodeLogonServer),
297 											nullSafe(unicodeUserName),
298 											nullSafe(unicodeDomainName),
299 											ntVersionStr);
300 									break;
301 							}
302 						}
303 					}
304 				}
305 			}
306 		}
307 	}
308 
309 	private static String nullSafe(String str) {
310 		return str != null && !str.isEmpty() ? "'" + str + "'" : "NULL";
311 	}
312 
313 	public static void main(String[] args) throws IOException, SignatureException {
314 		if (args.length == 0) {
315 			System.err.println("No arguments provided");
316 			System.exit(1);
317 		}
318 
319 		int positionalArgs = 0;
320 		String formatValue = "listing";
321 		boolean breakLoop = false;
322 		while (positionalArgs < args.length && !breakLoop) {
323 			switch (args[positionalArgs]) {
324 				case "--format":
325 					positionalArgs++;
326 					if (positionalArgs > args.length - 1)
327 						throw new IllegalArgumentException("Missing option value for '--format'");
328 					formatValue = args[positionalArgs++];
329 					break;
330 				case "--":
331 					positionalArgs++;
332 					breakLoop = true;
333 					break;
334 				default:
335 					breakLoop = true;
336 					break;
337 			}
338 		}
339 
340 		if (!formatValue.equals("listing") && !formatValue.equals("sql"))
341 			throw new IllegalArgumentException("Unsupported format value: " + formatValue);
342 
343 		final String format = formatValue;
344 		if (format.equals("sql")) {
345 			System.out.println("BEGIN TRANSACTION;");
346 			try (BufferedReader r = new BufferedReader(new InputStreamReader(
347 					ActiveDirectoryLdapPingerDumpPrinter.class.getResourceAsStream(
348 							"/net/sf/michaelo/activedirectory/ldapping/create-tables.sql"),
349 					StandardCharsets.UTF_8))) {
350 				r.lines().forEach(line -> System.out.println(line));
351 			}
352 		}
353 
354 		for (int i = positionalArgs; i < args.length; i++) {
355 			Path path = Paths.get(args[i]);
356 			if (Files.notExists(path)) {
357 				System.err.printf("Ignoring non-existing path '%s'%n", path);
358 				continue;
359 			}
360 			if (Files.isRegularFile(path)) {
361 				dumpFile(path, format);
362 			} else if (Files.isDirectory(path)) {
363 				Files.walk(path).filter(Files::isRegularFile).forEach(file -> {
364 					try {
365 						dumpFile(file, format);
366 					} catch (IOException e) {
367 						throw new UncheckedIOException(e);
368 					}
369 				});
370 			} else {
371 				System.err.printf("Ignoring unsupported path '%s'%n", path);
372 				continue;
373 			}
374 		}
375 
376 		if (format.equals("sql")) {
377 			System.out.println("COMMIT;");
378 		}
379 	}
