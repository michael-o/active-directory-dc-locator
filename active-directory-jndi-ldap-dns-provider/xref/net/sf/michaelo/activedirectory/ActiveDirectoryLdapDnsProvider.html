16  package net.sf.michaelo.activedirectory;
17  
18  import java.net.URI;
19  import java.net.URISyntaxException;
20  import java.util.Arrays;
21  import java.util.Collections;
22  import java.util.List;
23  import java.util.Map;
24  import java.util.Objects;
25  import java.util.Optional;
26  import java.util.logging.Level;
27  import java.util.logging.Logger;
28  import java.util.stream.Collectors;
29  
30  import javax.naming.ConfigurationException;
31  import javax.naming.NamingException;
32  import javax.naming.ldap.LdapName;
33  
34  import com.sun.jndi.ldap.spi.LdapDnsProvider;
35  import com.sun.jndi.ldap.spi.LdapDnsProviderResult;
36  
37  import net.sf.michaelo.activedirectory.DcLocatorRequest.Flag;
38  
39  /**
40   * An {@link LdapDnsProvider} for Active Directory. This implementation hooks into Sun's/Oracle's LDAP implementation of
41   * JNDI to autodiscover Active Directory servers via the DC locator process.
42   *
43   * <p>This provider receives all environment properties via JNDI. It recognizes the following properties:
44   *
45   * <ul>
46   *   <li>{@code net.sf.michaelo.activedirectory.readTimeout}: Read timeout for DC locator requests.
47   * </ul>
48   *
49   * <p>Note: This provider will return the input as-is if nothing can be located in the case that a simple host name has
50   * been provided.
51   */
52  @SuppressWarnings("restriction")
53  public class ActiveDirectoryLdapDnsProvider extends LdapDnsProvider {
54  
55  	public static final String READ_TIMEOUT_PROPERTY = "net.sf.michaelo.activedirectory.readTimeout";
56  
57  	private static final String LDAP_SCHEME = "ldap";
58  	private static final String LDAPS_SCHEME = "ldaps";
59  	private static final String GC_SCHEME = "gc";
60  	private static final String GCS_SCHEME = "gcs";
61  	private static final int GC_PORT = 3268;
62  	private static final int GCS_PORT = 3269;
63  
64  	private static final Logger LOGGER = Logger.getLogger(ActiveDirectoryLdapDnsProvider.class.getName());
65  
66  	@Override
67  	public Optional<LdapDnsProviderResult> lookupEndpoints(String url, Map<?, ?> env) throws NamingException {
68  		Objects.requireNonNull(url, "url cannot be null");
69  		Objects.requireNonNull(env, "env cannot be null");
70  
71  		URI ldapUri = null;
72  		try {
73  			ldapUri = URI.create(url);
74  		} catch (IllegalArgumentException e) {
75  			NamingException ne = new ConfigurationException("URL '" + url + "' is invalid");
76  			ne.initCause(e);
77  			throw ne;
78  		}
79  
80  		String val = (String) env.get(READ_TIMEOUT_PROPERTY);
81  		int readTimeout = (val == null) ? -1 : Integer.parseInt(val);
82  
83  		ActiveDirectoryDcLocator locator = new ActiveDirectoryDcLocator();
84  		DcLocatorRequest request = new DcLocatorRequest();
85  		request.addFlag(Flag.DS_ONLY_LDAP_NEEDED);
86  		request.addFlag(Flag.DS_RETURN_DNS_NAME);
87  		request.addFlag(Flag.DS_TRY_NEXTCLOSEST_SITE);
88  		request.setReadTimeout(readTimeout);
89  
90  		String scheme = ldapUri.getScheme();
91  		String domainName = ldapUri.getHost();
92  		String path = ldapUri.getRawPath();
93  		String baseDnStr = StringUtils.isNotEmpty(path) ? path.replaceFirst("/", "") : "";
94  		int port = ldapUri.getPort();
95  
96  		if (StringUtils.isEmpty(domainName) && StringUtils.isNotEmpty(baseDnStr)) {
97  			LdapName baseDn = new LdapName(baseDnStr);
98  			List<String> dcs = baseDn.getRdns().stream()
99  					.filter(rdn -> rdn.getType().equalsIgnoreCase("DC"))
100 					.map(rdn -> (String) rdn.getValue())
101 					.collect(Collectors.toList());
102 			Collections.reverse(dcs);
103 			domainName = String.join(".", dcs);
104 		}
105 		request.setDomainName(domainName);
106 
107 		boolean isGcServerRequired = false;
108 		if (scheme.equalsIgnoreCase(LDAP_SCHEME) && port == GC_PORT
109 				|| scheme.equalsIgnoreCase(LDAPS_SCHEME) && port == GCS_PORT) isGcServerRequired = true;
110 		else if (scheme.equalsIgnoreCase(GC_SCHEME)) {
111 			isGcServerRequired = true;
112 			scheme = LDAP_SCHEME;
113 			port = port == -1 ? GC_PORT : port;
114 		} else if (scheme.equalsIgnoreCase(GCS_SCHEME)) {
115 			isGcServerRequired = true;
116 			scheme = LDAPS_SCHEME;
117 			port = port == -1 ? GCS_PORT : port;
118 		}
119 		if (isGcServerRequired) request.addFlag(Flag.DS_GC_SERVER_REQUIRED);
120 
121 		try {
122 			LOGGER.fine(() -> String.format("Locating a server for '%s'", url));
123 			DomainControllerInfo dcInfo = locator.locate(request);
124 			LOGGER.fine(() -> String.format("Successfully located a server for '%s': %s", url, dcInfo));
125 			try {
126 				URI endpointUri = new URI(
127 						scheme,
128 						null,
129 						dcInfo.getDomainControllerName(),
130 						port,
131 						path,
132 						ldapUri.getQuery(),
133 						ldapUri.getFragment());
134 				String _domainName = isGcServerRequired ? dcInfo.getDnsForestName() : dcInfo.getDomainName();
135 				String endpoint = endpointUri.toASCIIString();
136 				LOGGER.fine(() -> String.format("Returning domainName '%s' with endpoint '%s'", _domainName, endpoint));
137 				// This is poorly documented, it conflates domain (names) with actual endpoint hostnames, so which is
138 				// required?
139 				return Optional.of(new LdapDnsProviderResult(_domainName, Arrays.asList(endpoint)));
140 			} catch (URISyntaxException e) {
141 				NamingException ne = new NamingException("Failed to construct endpoint URL");
142 				ne.initCause(e);
143 				throw ne;
144 			}
145 		} catch (NamingException e) {
146 			LOGGER.log(Level.FINE, e, () -> "Failed to locate a server, returning original as fallback...");
147 			return Optional.of(new LdapDnsProviderResult("", Arrays.asList(url)));
148 		}
149 	}
150 }
